The class structure:
```typescript
import { ENTITY_TYPE } from '@openvaa/data';
import { UniversalDataWriter } from '$lib/api/base/universalDataWriter';
import { supabaseAdapterMixin } from '../supabaseAdapter';
import type { DataApiActionResult } from '$lib/api/base/actionResult.type';
import { constants } from '$lib/utils/constants';
import { toDataObject } from '../utils/toDataObject';
import { parseStoredImage } from '../utils/storageUrl';
import type {
  BasicUserData, CandidateUserData, CheckRegistrationData, DWReturnType,
  GetCandidateUserDataOptions, InsertJobResultOptions, LocalizedAnswers,
  LocalizedCandidateData, SetAnswersOptions, SetPropertiesOptions,
  SetQuestionOptions, WithAuth
} from '$lib/api/base/dataWriter.type';

export class SupabaseDataWriter extends supabaseAdapterMixin(UniversalDataWriter) {
  // AUTH METHODS
  _login, _logout, logout (public override), _requestForgotPasswordEmail,
  _resetPassword, _setPassword

  // REGISTRATION METHODS
  _preregister (Edge Function), _checkRegistrationKey (throws not supported), _register

  // USER DATA METHODS
  _getBasicUserData, _getCandidateUserData

  // ANSWER/PROPERTY METHODS
  _setAnswers (with File upload to Storage), _updateEntityProperties

  // ADMIN METHODS (kept for abstract contract, with TODO)
  _updateQuestion (merge_custom_data RPC), _insertJobResult (admin_jobs insert)
}
```

Key implementation details:
- `_login`: calls `this.supabase.auth.signInWithPassword({ email: username, password })`
- `_logout`: calls fetch to server-side logout endpoint (browser), then `this.supabase.auth.signOut({ scope: 'local' })`
- `logout` (public override): skips UniversalDataWriter's dual POST+backendLogout pattern, calls `this._logout()` directly
- `_preregister`: resolves `project_id` from election, invokes `invite-candidate` Edge Function
- `_getBasicUserData`: gets session, decodes JWT access_token for `user_roles` claim, determines role
- `_getCandidateUserData`: calls `get_candidate_user_data` RPC, optionally loads nominations
- `_setAnswers`: handles File upload to `public-assets` Storage bucket, calls `upsert_answers` RPC
- `_updateEntityProperties`: updates candidates table, handles image upload to Storage
