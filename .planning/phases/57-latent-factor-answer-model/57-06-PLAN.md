```typescript
import { COORDINATE } from '@openvaa/core';
import type { Enums, TablesInsert } from '@openvaa/supabase-types';
import type { Ctx } from '../../ctx';
import type { LatentHooks, LoadingMatrix } from './latentTypes';
import { defaultRandomValidEmit } from '../answers';
import { boxMuller } from './gaussian';

type QuestionType = Enums<'question_type'>;

/**
 * defaultProject — GEN-06f built-in default.
 *
 * Projects each candidate's latent `position` through the loading matrix,
 * adds Gaussian noise (D-57-11, `noiseStdDev = 0` short-circuits), and maps
 * the scalar `z` to a valid answer per question type.
 *
 * Per D-57-08, ordinal mapping uses the inverse of `@openvaa/core`'s
 * `normalizeCoordinate` formula. We compute `z` in the COORDINATE range
 * [-0.5, +0.5] and back out the raw `normalizableValue`, then find the
 * closest choice. NO hand-rolled bucket tables.
 *
 * Per D-57-09, categorical types use per-choice loading vectors (dense N(0,1))
 * sampled ONCE when the question is first seen. Single-choice: argmax;
 * multi-choice: include choices with positive dot product (≥1 guarantee).
 *
 * Per D-57-10, non-ordinal/non-choice types fall through to
 * `defaultRandomValidEmit` — latent projection has no semantic meaning for
 * free-form text, numbers, dates, booleans, or images.
 */
export function defaultProject(
  position: Array<number>,
  loadings: LoadingMatrix,
  questions: ReadonlyArray<TablesInsert<'questions'>>,
  noiseStdDev: number,
  ctx: Ctx
): Record<string, { value: unknown; info?: unknown }> {
  const out: Record<string, { value: unknown; info?: unknown }> = {};

  // Per-question categorical-choice loadings cache (keyed by qExtId). Sampled
  // lazily the first time we need them; deterministic via ctx.faker.
  const choiceLoadingsCache: Record<string, Array<Array<number>>> = {};

  for (const q of questions) {
    const qExtId = q.external_id;
    if (!qExtId) continue;

    const type = q.type as QuestionType;
    switch (type) {
      case 'singleChoiceOrdinal': {
        const qLoading = loadings[qExtId] ?? [];
        const z = computeZ(position, qLoading, noiseStdDev, ctx);
        out[qExtId] = { value: mapOrdinal(q, z) };
        break;
      }
      case 'singleChoiceCategorical': {
        out[qExtId] = { value: mapSingleCategorical(q, position, ctx, choiceLoadingsCache, qExtId) };
        break;
      }
      case 'multipleChoiceCategorical': {
        out[qExtId] = { value: mapMultiCategorical(q, position, ctx, choiceLoadingsCache, qExtId) };
        break;
      }
      case 'text':
      case 'multipleText':
      case 'number':
      case 'boolean':
      case 'date':
      case 'image': {
        // D-57-10 fallback — reuse Phase 56 stub.
        const fallback = defaultRandomValidEmit(
          {} as TablesInsert<'candidates'>,
          [q],
          ctx
        );
        out[qExtId] = fallback[qExtId];
        break;
      }
      default: {
        const _exhaustive: never = type;
        void _exhaustive;
        out[qExtId] = { value: null };
      }
    }
  }
  return out;
}

// --- Helpers ---

function computeZ(position: Array<number>, qLoading: Array<number>, noiseStdDev: number, ctx: Ctx): number {
  let z = 0;
  const L = Math.min(position.length, qLoading.length);
  for (let d = 0; d < L; d++) z += position[d] * qLoading[d];
  z += boxMuller(ctx.faker, 0, noiseStdDev);
  return z;
}

/** D-57-08: inverse-normalize z through @openvaa/core COORDINATE semantics. */
function mapOrdinal(q: TablesInsert<'questions'>, z: number): string | null {
  const choices = extractOrdinalChoices(q);
  if (choices.length === 0) return null;
  const zClipped = Math.max(COORDINATE.Min, Math.min(COORDINATE.Max, z));
  const values = choices.map((c) => c.normalizableValue);
  const vmin = Math.min(...values);
  const vmax = Math.max(...values);
  // Handle the degenerate single-value-range case.
  if (vmin === vmax) return choices[0].id;
  const targetValue = vmin + ((zClipped - COORDINATE.Min) / COORDINATE.Extent) * (vmax - vmin);
  let best = choices[0];
  let bestDiff = Math.abs(best.normalizableValue - targetValue);
  for (const c of choices) {
    const diff = Math.abs(c.normalizableValue - targetValue);
    if (diff < bestDiff) {
      bestDiff = diff;
      best = c;
    }
  }
  return best.id;
}

function mapSingleCategorical(
  q: TablesInsert<'questions'>,
  position: Array<number>,
  ctx: Ctx,
  cache: Record<string, Array<Array<number>>>,
  qExtId: string
): string | null {
  const choices = extractChoiceIds(q);
  if (choices.length === 0) return null;
  const cls = getOrSampleChoiceLoadings(cache, qExtId, choices.length, position.length, ctx);
  let bestIdx = 0;
  let bestDot = -Infinity;
  for (let i = 0; i < choices.length; i++) {
    const d = dot(position, cls[i]);
    if (d > bestDot) {
      bestDot = d;
      bestIdx = i;
    }
  }
  return choices[bestIdx];
}

function mapMultiCategorical(
  q: TablesInsert<'questions'>,
  position: Array<number>,
  ctx: Ctx,
  cache: Record<string, Array<Array<number>>>,
  qExtId: string
): Array<string> {
  const choices = extractChoiceIds(q);
  if (choices.length === 0) return [];
  const cls = getOrSampleChoiceLoadings(cache, qExtId, choices.length, position.length, ctx);
  const picked: Array<string> = [];
  let bestIdx = 0;
  let bestDot = -Infinity;
  for (let i = 0; i < choices.length; i++) {
    const d = dot(position, cls[i]);
    if (d > 0) picked.push(choices[i]);
    if (d > bestDot) {
      bestDot = d;
      bestIdx = i;
    }
  }
  // D-57-09 / S-4: guarantee ≥ 1 selection — if no dots positive, take the argmax.
  if (picked.length === 0) picked.push(choices[bestIdx]);
  return picked;
}

function getOrSampleChoiceLoadings(
  cache: Record<string, Array<Array<number>>>,
  qExtId: string,
  numChoices: number,
  dims: number,
  ctx: Ctx
): Array<Array<number>> {
  const existing = cache[qExtId];
  if (existing) return existing;
  const cls = Array.from({ length: numChoices }, () =>
    Array.from({ length: dims }, () => boxMuller(ctx.faker, 0, 1))
  );
  cache[qExtId] = cls;
  return cls;
}

function dot(a: Array<number>, b: Array<number>): number {
  let s = 0;
  const L = Math.min(a.length, b.length);
  for (let i = 0; i < L; i++) s += a[i] * b[i];
  return s;
}

// S-5 pattern from emitters/answers.ts:129-140 — mirrored with normalizableValue extraction per RESEARCH lines 863-884.
function extractOrdinalChoices(q: TablesInsert<'questions'>): Array<{ id: string; normalizableValue: number }> {
  const choices = q.choices;
  if (!Array.isArray(choices)) return [];
  return choices
    .map((c) => {
      if (c && typeof c === 'object' && 'id' in c) {
        const id = String((c as { id: unknown }).id);
        const nv =
          typeof (c as { normalizableValue?: unknown }).normalizableValue === 'number'
            ? (c as { normalizableValue: number }).normalizableValue
            : Number.isFinite(Number(id))
            ? Number(id)
            : NaN;
        if (!Number.isFinite(nv)) return null;
        return { id, normalizableValue: nv };
      }
      return null;
    })
    .filter((v): v is { id: string; normalizableValue: number } => v !== null);
}

function extractChoiceIds(q: TablesInsert<'questions'>): Array<string> {
  const choices = q.choices;
  if (!Array.isArray(choices)) return [];
  return choices
    .map((c) => {
      if (c && typeof c === 'object' && 'id' in c) {
        const id = (c as { id: unknown }).id;
        if (typeof id === 'string') return id;
      }
      return null;
    })
    .filter((v): v is string => v !== null);
}

const _typecheckProject: NonNullable<LatentHooks['project']> = defaultProject;
void _typecheckProject;
```

- The per-choice categorical loading cache is a local `Record<qExtId, Array<Array<number>>>` created AT FUNCTION ENTRY — NOT at module scope, and NOT on ctx. Same candidate crossing `mapSingleCategorical` for `q_0` three times in a single call (impossible in practice — questions appear once in the array, but semantically the cache is per-project-invocation) sees the same loadings. This is a defensive guard; in the live emitter shell (Plan 07), `defaultProject` is called once per candidate, so the cache is per-candidate. If that's undesired behavior (same question → different choice loadings per candidate), the CLOSURE-CACHED version lives in Plan 07's emitter shell. Actually, correction: re-read D-57-09 + RESEARCH. Per-choice loadings should be sampled ONCE per pipeline run (matching the "loadings cached in closure" pattern). Plan 07's `latentAnswerEmitter` caches `dims`/`centroids`/`loadings` once. The question-loading matrix produced by `defaultLoadings` (Plan 05) is the PER-QUESTION vector — it does NOT include per-CHOICE vectors. The per-choice vectors are an implementation detail of `defaultProject` that must also be memoized at pipeline-run scope.

Therefore the choice-loading cache CANNOT be created inside `defaultProject` (it would be per-call, per-candidate — wrong). It must be pulled into the closure that Plan 07's emitter shell owns, OR stored on `ctx` in a dedicated sub-field, OR the cache must live at module scope (anti-pattern per RESEARCH Anti-Patterns — mutates global).

**Resolution:** Store the cache on `ctx` as a non-mutating dedicated field — but `Ctx` type is fixed by Plan 01 to `LatentHooks` only. Alternative: `defaultProject` accepts an optional cache argument passed by `latentAnswerEmitter`. BUT the `LatentHooks.project` signature does NOT include a cache parameter, so user-supplied project hooks cannot access it.

**Cleanest resolution:** `defaultProject` accepts the cache implicitly via module-level `WeakMap<Ctx, ChoiceLoadingCache>` keyed by the ctx object reference. Same ctx → same cache. Fresh ctx (e.g. new pipeline run) → fresh cache. No global mutation visible outside the module.

Implementation:

```typescript
// Module-scoped WeakMap keyed by ctx identity. One pipeline run = one ctx = one cache.
// Since each ctx is built fresh in `buildCtx(template)`, no cross-run bleed.
const CHOICE_LOADINGS: WeakMap<Ctx, Record<string, Array<Array<number>>>> = new WeakMap();

function getChoiceLoadings(ctx: Ctx): Record<string, Array<Array<number>>> {
  let cache = CHOICE_LOADINGS.get(ctx);
  if (!cache) {
    cache = {};
    CHOICE_LOADINGS.set(ctx, cache);
  }
  return cache;
}
```

Then `mapSingleCategorical` / `mapMultiCategorical` call `getChoiceLoadings(ctx)` instead of passing `cache` as a parameter. This preserves:
- Determinism (same ctx → same cache → same choice loadings).
