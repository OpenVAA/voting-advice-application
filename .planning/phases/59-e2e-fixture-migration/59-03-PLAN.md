```typescript
#!/usr/bin/env tsx
/**
 * Compare two Playwright JSON reports (baseline vs post-swap) per D-59-04.
 *
 * Rules:
 *   - Every test in baseline.pass MUST be in post.pass (regression if not).
 *   - Every test in baseline.cascade MAY be in post.pass or post.cascade — flex accepted.
 *   - Tests in baseline.data_race (the 19-test pool) MAY flake differently post.
 *   - No test outside baseline.data_race may newly enter the data_race or fail set.
 *
 * "Did not run" / timedOut is counted as a failure per D-59-04 + stored memory
 * feedback_e2e_did_not_run.md.
 *
 * Exit: 0 if parity satisfied, 1 if regression detected.
 * Usage:
 *   tsx diff-playwright-reports.ts <baseline.json> <post-swap.json>
 */

import { readFileSync } from 'node:fs';
import { parseArgs } from 'node:util';

// Sourced from baseline/summary.md "Data-race Failing Tests" section at
// authoring time. Embedded inline per 59-PATTERNS.md §"Diff script" note
// (single source of truth; re-parsing summary.md markdown is brittle).
//
// Executor: after Plan 01 produces baseline/summary.md, read its data-race
// section and paste the full "<suite.file> › <spec.title>" list below, one
// entry per array element. The list length should match the N_dr count
// printed in summary.md.
const DATA_RACE_TESTS: ReadonlyArray<string> = [
  // <PASTE 19 test identifiers here from baseline/summary.md, one per line>
];

interface PlaywrightTestResult {
  status?: 'passed' | 'failed' | 'skipped' | 'timedOut' | 'interrupted';
  ok?: boolean;
  error?: { message?: string };
}
interface PlaywrightTest { results?: Array<PlaywrightTestResult> }
interface PlaywrightSpec { title?: string; tests?: Array<PlaywrightTest> }
interface PlaywrightSuite {
  title?: string;
  file?: string;
  specs?: Array<PlaywrightSpec>;
  suites?: Array<PlaywrightSuite>;
}
interface PlaywrightReport { suites?: Array<PlaywrightSuite> }

type TestStatus = 'pass' | 'fail' | 'cascade';

interface FlatTest {
  id: string;    // "<file> › <title>"
  status: TestStatus;
  rawStatus: string;
  errorMessage?: string;
}

function flattenReport(report: PlaywrightReport): Array<FlatTest> {
  const out: Array<FlatTest> = [];
  const walk = (suites: Array<PlaywrightSuite> | undefined, pathPrefix: string) => {
    for (const suite of suites ?? []) {
      const file = suite.file ?? suite.title ?? pathPrefix;
      for (const spec of suite.specs ?? []) {
        for (const t of spec.tests ?? []) {
          const raw = t.results?.[0]?.status ?? 'skipped';
          const err = t.results?.[0]?.error?.message ?? '';
          const isCascade = /did not run|setup.*failed|dependency.*failed/i.test(err) || raw === 'skipped';
          const status: TestStatus = raw === 'passed' ? 'pass' : isCascade ? 'cascade' : 'fail';
          out.push({ id: `${file} › ${spec.title ?? ''}`, status, rawStatus: raw, errorMessage: err });
        }
      }
      if (suite.suites && suite.suites.length > 0) walk(suite.suites, `${pathPrefix}${file}/`);
    }
  };
  walk(report.suites, '');
  return out;
}

export interface DiffResult {
  satisfied: boolean;
  regressions: Array<{ id: string; from: TestStatus; to: TestStatus; reason: string }>;
  summary: {
    baseline: { pass: number; fail: number; cascade: number };
    post: { pass: number; fail: number; cascade: number };
    dataRacePoolSize: number;
  };
}

export function diffReports(baseline: PlaywrightReport, post: PlaywrightReport): DiffResult {
  const baseFlat = flattenReport(baseline);
  const postFlat = flattenReport(post);
  const baseById = new Map(baseFlat.map((t) => [t.id, t]));
  const postById = new Map(postFlat.map((t) => [t.id, t]));
  const dataRace = new Set(DATA_RACE_TESTS);
  const regressions: DiffResult['regressions'] = [];

  for (const b of baseFlat) {
    const p = postById.get(b.id);
    const postStatus: TestStatus = p?.status ?? 'cascade'; // missing post = did-not-run = cascade
    if (b.status === 'pass' && postStatus !== 'pass') {
      regressions.push({ id: b.id, from: 'pass', to: postStatus, reason: 'baseline passing test regressed' });
    } else if (b.status !== 'pass' && !dataRace.has(b.id) && postStatus === 'fail') {
      // cascade-or-fail moving to fail is fine IF it was cascade before (flex).
      // But if it was PASS before (handled above) it's a regression. If it was
      // fail-outside-data-race before, a new fail post-swap is still "fail" — no new regression.
      // Only cascade->fail outside the pool is a concern — document but don't block (same family of failures).
    }
  }

  // Detect the inverse: a test that wasn't in baseline appears failing post-swap.
  for (const p of postFlat) {
    if (!baseById.has(p.id) && (p.status === 'fail' || p.status === 'cascade')) {
      regressions.push({ id: p.id, from: 'cascade', to: p.status, reason: 'new test appeared in failing/cascade state' });
    }
  }

  const tally = (xs: Array<FlatTest>) => ({
    pass: xs.filter((x) => x.status === 'pass').length,
    fail: xs.filter((x) => x.status === 'fail').length,
    cascade: xs.filter((x) => x.status === 'cascade').length
  });

  return {
    satisfied: regressions.length === 0,
    regressions,
    summary: {
      baseline: tally(baseFlat),
      post: tally(postFlat),
      dataRacePoolSize: dataRace.size
    }
  };
}

const isDirect =
  typeof process.argv[1] === 'string' &&
  (process.argv[1].endsWith('diff-playwright-reports.ts') ||
   process.argv[1].endsWith('diff-playwright-reports.js'));

if (isDirect) {
  const { positionals } = parseArgs({ allowPositionals: true, strict: false, options: {} });
  if (positionals.length !== 2) {
    process.stderr.write('Usage: tsx diff-playwright-reports.ts <baseline.json> <post.json>\n');
    process.exit(2);
  }
  try {
    const base = JSON.parse(readFileSync(positionals[0], 'utf8')) as PlaywrightReport;
    const post = JSON.parse(readFileSync(positionals[1], 'utf8')) as PlaywrightReport;
    const result = diffReports(base, post);
    process.stdout.write(
      `Baseline: ${result.summary.baseline.pass}p / ${result.summary.baseline.fail}f / ${result.summary.baseline.cascade}c\n`
    );
    process.stdout.write(
      `Post:     ${result.summary.post.pass}p / ${result.summary.post.fail}f / ${result.summary.post.cascade}c\n`
    );
    process.stdout.write(`Data-race pool: ${result.summary.dataRacePoolSize} tests (filtered from regression analysis).\n`);
    if (result.regressions.length === 0) {
      process.stdout.write('PARITY GATE: PASS — no regressions detected per D-59-04.\n');
      process.exit(0);
    } else {
      process.stdout.write(`PARITY GATE: FAIL — ${result.regressions.length} regression(s):\n`);
      for (const r of result.regressions) {
        process.stdout.write(`  - [${r.from} -> ${r.to}] ${r.id} (${r.reason})\n`);
      }
      process.exit(1);
    }
  } catch (err) {
    const m = (err as Error)?.message ?? String(err);
    process.stderr.write(`Error: ${m}\n`);
    process.exit(2);
  }
```
