# Plan 33-04: Protected Layout Guards and Server-Side Auth

<objective>
Wire session-based authentication into the protected layout guard, the candidate server layout, and the hooks.server.ts auth redirect handler. The protected layout checks `page.data.session` instead of `page.data.token`. The candidate server layout provides `session` from `safeGetSession()`. The candidateAuthHandle in hooks uses session-based checks.
</objective>

<must_haves>
- candidate/+layout.server.ts returns `{ session }` from `safeGetSession()` instead of `{ token }` from cookie
- Protected layout loader checks `(await parent()).session` instead of `(await parent()).token`
- Protected layout passes `authToken: ''` to all DataWriter calls
- hooks.server.ts candidateAuthHandle uses `safeGetSession()` for session check instead of `cookies.get(AUTH_TOKEN_KEY)`
- hooks.server.ts no longer imports `AUTH_TOKEN_KEY`
</must_haves>

Update `apps/frontend/src/routes/candidate/(protected)/+layout.ts` to use session-based auth instead of token-based. The file stays as a universal loader (`.ts` not `.server.ts`) because it uses the adapter-agnostic dataWriter/dataProvider singletons (not per-request Supabase instances).

Replace the entire contents with:

```typescript
/**
 * Load the data for a logged-in candidate.
 * - Verify session-based authentication via parent data
 * - Load user data via DataWriter
 * - Load relevant question data based on the candidate's nominations via DataProvider
 *
 * Redirects to login with an error message if the user is not logged in.
 */

import { redirect } from '@sveltejs/kit';
import { dataProvider as dataProviderPromise } from '$lib/api/dataProvider';
import { dataWriter as dataWriterPromise } from '$lib/api/dataWriter';
import { getLocale } from '$lib/paraglide/runtime';
import { logDebugError } from '$lib/utils/logger';
import { removeDuplicates } from '$lib/utils/removeDuplicates';
import { buildRoute } from '$lib/utils/route';
import type { Id } from '@openvaa/core';
import type { CandidateLoginError } from '$candidate/utils/loginError';

export async function load({ fetch, parent }) {
  const lang = getLocale();

  // Init dataWriter
  const dataWriter = await dataWriterPromise;
  dataWriter.init({ fetch });

  // Check for valid session (provided by candidate/+layout.server.ts)
  const { session } = await parent();
  if (!session)
    return redirect(
      307,
      buildRoute({
        route: 'CandAppLogin',
        locale: lang
      })
    );

  // Get user data -- authToken is '' because Supabase uses cookie-based sessions
  const userData = await dataWriter.getCandidateUserData({ authToken: '', loadNominations: true }).catch((e) => {
    logDebugError(`Error fetching user data: ${e?.message ?? 'No error message'}`);
    return undefined;
  });
  if (!userData) return await handleError('loginFailed');

  // Check that the data is valid and the user is a candidate
  const {
    user: { role },
    candidate,
    nominations: { nominations }
  } = userData;
  if (role !== 'candidate') return await handleError('userNotAuthorized');
  if (!candidate) return await handleError('userNoCandidate');

  // Parse the election and constituency ids
  let electionId = new Array<Id>();
  let constituencyId = new Array<Id>();
  if (nominations) {
    electionId = removeDuplicates(nominations.map((n) => n.electionId));
    constituencyId = removeDuplicates(nominations.map((n) => n.constituencyId));
  }
  if (!electionId.length || !constituencyId.length) return await handleError('candidateNoNomination');

  // Get question data
  const dataProvider = await dataProviderPromise;
  dataProvider.init({ fetch });

  // Await questionData to avoid SvelteKit streaming issues in dev mode.
  // Without await, the Promise is streamed to the client, but Vite's dev
  // server may not complete the stream properly after a form action redirect.
  const questionData = await dataProvider
    .getQuestionData({
      electionId,
      locale: lang
    })
    .catch((e) => e);

  return {
    questionData,
    candidateUserData: userData
  };

  /**
   * Call logout and redirect to the login page with an error message.
   */
  async function handleError(error: CandidateLoginError): Promise<void> {
    await dataWriter
      .logout({ authToken: '' })
      .catch((e) => logDebugError(`[Candidate App protected layout] Error logging out: ${e?.message ?? '-'}`));
    redirect(
      307,
      buildRoute({
        route: 'CandAppLogin',
        locale: lang,
        errorMessage: error
      })
    );
  }
}
```

Key changes from current:
- `const authToken = (await parent()).token` -> `const { session } = await parent()`
- `if (!authToken)` -> `if (!session)`
- `{ authToken, loadNominations: true }` -> `{ authToken: '', loadNominations: true }`
- `{ authToken: token }` in getCandidateUserData -> `{ authToken: '' }`
- `.logout({ authToken: authToken ?? '' })` -> `.logout({ authToken: '' })`
</action>

    * Candidate auth redirect handler (Supabase session).
    * Redirects logged-in users away from login and unauthenticated users to login.
    */
   const candidateAuthHandle: Handle = async ({ event, resolve }) => {
     const { url, route } = event;
     const locale = getLocale();
     const pathname = url.pathname;

     // Skip non-route and API requests
     if (route?.id == null || pathname.startsWith(NORMALIZED_API_ROOT)) {
       return resolve(event);
     }

     // Handle candidate auth redirects
     if (pathname.includes('/candidate')) {
       const { session } = await event.locals.safeGetSession();
       if (session && pathname.endsWith('candidate/login')) {
         redirect(303, `/${locale}/candidate`);
       }
       if (!session && route.id.includes('(protected)')) {
         const cleanPath = pathname.replace(new RegExp(`^/${locale}`), '');
         redirect(303, `/${locale}/candidate/login?redirectTo=${cleanPath.substring(1)}`);
       }
     }

     return resolve(event);
   };
