Sub-module conversion pattern (getter functions for reactive inputs):
```typescript
// BEFORE: accepts Readable<T>, returns Readable<T>
export function someStore({ dataRoot }: { dataRoot: Readable<DataRoot> }): Readable<Result> {
  return derived(dataRoot, (dr) => compute(dr));
}

// AFTER: accepts () => T getters, returns reactive value via $derived
export function someStore({ dataRoot }: { dataRoot: () => DataRoot }): { readonly value: Result } {
  const value = $derived.by(() => compute(dataRoot()));
  return { get value() { return value; } };
}
// OR simpler: return the $derived directly if the calling context handles it
```
