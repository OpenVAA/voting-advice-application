import type { TablesInsert, Enums } from '@openvaa/supabase-types';
import type { Ctx, Fragment } from '../types';

export type QuestionCategoriesFragment = Fragment<TablesInsert<'question_categories'>>;

export class QuestionCategoriesGenerator {
  constructor(private ctx: Ctx) {}

  defaults(_ctx: Ctx): QuestionCategoriesFragment {
    return { count: 2 };  // e.g. "Economy", "Environment"
  }

  generate(fragment: QuestionCategoriesFragment): Array<TablesInsert<'question_categories'>> {
    const { faker, projectId, externalIdPrefix } = this.ctx;
    const rows: Array<TablesInsert<'question_categories'>> = [];

    for (const fx of fragment.fixed ?? []) {
      rows.push({ ...fx, external_id: `${externalIdPrefix}${fx.external_id}`, project_id: fx.project_id ?? projectId });
    }

    for (let i = 0; i < (fragment.count ?? 0); i++) {
      rows.push({
        external_id: `${externalIdPrefix}cat_${String(i).padStart(2, '0')}`,
        project_id: projectId,
        name: { en: `${faker.word.adjective()} Category ${i + 1}` } as unknown as TablesInsert<'question_categories'>['name'],
        category_type: 'opinion' satisfies Enums<'category_type'>,
        sort_order: i,
        is_generated: true
        // _elections sentinel added by Plan 07's post-topo pass
      });
    }
    return rows;
  }
}

Questions require `category_id` — supplied via `category: { external_id }` ref object. Each question has a `type` and for ordinal/categorical types, non-null `choices` with ≥2 entries.

```ts
import type { TablesInsert, Enums } from '@openvaa/supabase-types';
import type { Ctx, Fragment } from '../types';

export type QuestionsFragment = Fragment<TablesInsert<'questions'>>;

// Questions require `category: { external_id }` ref, which isn't on the Insert type.
type QuestionRow = TablesInsert<'questions'> & { category?: { external_id: string } };

/**
 * Standard Phase 56 Likert choices (5-point scale). The DB trigger
 * validate_question_choices requires ≥2 entries each with an `id` key.
 * Stored as JSONB; bulk_import passes through (Json type on the column).
 */
const LIKERT_5: Array<{ id: string; label: { en: string } }> = [
  { id: '1', label: { en: 'Strongly disagree' } },
  { id: '2', label: { en: 'Disagree' } },
  { id: '3', label: { en: 'Neutral' } },
  { id: '4', label: { en: 'Agree' } },
  { id: '5', label: { en: 'Strongly agree' } }
];

/**
 * Simple categorical choices (for singleChoiceCategorical / multipleChoiceCategorical).
 * Phase 56 emits a minimal 3-option set; Phase 58 templates can supply richer ones.
 */
const CATEGORICAL_3: Array<{ id: string; label: { en: string } }> = [
  { id: 'a', label: { en: 'Option A' } },
  { id: 'b', label: { en: 'Option B' } },
  { id: 'c', label: { en: 'Option C' } }
];

export class QuestionsGenerator {
  constructor(private ctx: Ctx) {}

  defaults(_ctx: Ctx): QuestionsFragment {
    return { count: 4 };  // Enough questions to exercise the answer emitter & matching across a few dimensions
  }

  generate(fragment: QuestionsFragment): Array<TablesInsert<'questions'>> {
    const { faker, projectId, externalIdPrefix, refs } = this.ctx;
    const rows: Array<QuestionRow> = [];

    for (const fx of fragment.fixed ?? []) {
      rows.push({ ...fx, external_id: `${externalIdPrefix}${fx.external_id}`, project_id: fx.project_id ?? projectId });
    }

    // Pick a category external_id to attach; rotate across available categories so multiple questions
    // land in each. If no categories exist yet (edge case: template supplies questions but no categories),
    // generator emits the rows WITHOUT `category` — bulk_import will fail loudly with a clear error.
    const categoryExtIds = refs.question_categories.map((c) => c.external_id);

    for (let i = 0; i < (fragment.count ?? 0); i++) {
      // Phase 56 emits a mix of question types so the answer emitter exercises every branch.
      // Rotate deterministically (via i % types.length) so seeded runs are reproducible.
      const type = (['singleChoiceOrdinal', 'boolean', 'singleChoiceCategorical', 'text'] satisfies Enums<'question_type'>[])[i % 4];

      const row: QuestionRow = {
        external_id: `${externalIdPrefix}q_${String(i).padStart(3, '0')}`,
        project_id: projectId,
        type: type,
        text: { en: faker.lorem.sentence({ min: 6, max: 12 }) + '?' } as unknown as TablesInsert<'questions'>['text'],
        allow_open: true,
        required: true,
        sort_order: i
      };

      // choices required for ordinal/categorical types (RESEARCH §4.13 validate_question_choices trigger)
      if (type === 'singleChoiceOrdinal') {
        row.choices = LIKERT_5 as unknown as TablesInsert<'questions'>['choices'];
      } else if (type === 'singleChoiceCategorical' || type === 'multipleChoiceCategorical') {
        row.choices = CATEGORICAL_3 as unknown as TablesInsert<'questions'>['choices'];
      }

      if (categoryExtIds.length > 0) {
        const catExtId = categoryExtIds[i % categoryExtIds.length];
        row.category = { external_id: catExtId };
      }

      rows.push(row);
    }

    return rows as Array<TablesInsert<'questions'>>;
  }
}
```

```ts
import type { TablesInsert } from '@openvaa/supabase-types';
import type { Ctx, Fragment } from '../types';
import { defaultRandomValidEmit } from '../emitters/answers';

export type CandidatesFragment = Fragment<TablesInsert<'candidates'>>;

// Candidates carry two sentinel / ref fields not in the Insert type:
//   - `organization: { external_id }` — bulk_import resolves to organization_id (migration line 2624)
//   - `answersByExternalId: { [qExtId]: { value } }` — stripped by bulk_import, read by importAnswers
type CandidateRow = TablesInsert<'candidates'> & {
  organization?: { external_id: string };
  answersByExternalId?: Record<string, { value: unknown; info?: unknown }>;
};

/**
 * CandidatesGenerator — emits candidate rows with organization refs and random-valid answers.
 *
 * D-26: `ctx` captured at construction. `generate(fragment)` operates on `this.ctx`.
 * D-27 seam: `const emit = ctx.answerEmitter ?? defaultRandomValidEmit` — Phase 57's
 * latent-factor emitter drops into `ctx.answerEmitter` without changing this file.
 * D-19: Phase 56 default = random-valid-per-question-type stub.
 * D-20: shape-valid ONLY. Subdimension / MISSING_VALUE logic stays in @openvaa/matching.
 *
 * Ref dependencies (requires Plan 07's pipeline to run generators in D-06 topo order):
 *   - `ctx.refs.organizations` must be populated (candidates pick a party)
 *   - `ctx.refs.questions` must be populated (answer emitter needs to know which questions exist)
 *
 * If either ref is empty at generate time, candidates are emitted without the
 * corresponding field — the writer then fails loudly at bulk_import or importAnswers
 * with a specific missing-ref error. Failing fast is better than silently producing
 * orphan candidates.
 */
export class CandidatesGenerator {
  constructor(private ctx: Ctx) {}

  defaults(_ctx: Ctx): CandidatesFragment {
    return { count: 8 };  // Enough candidates per party for visible matching patterns (Phase 57 tunes this upward)
  }

  generate(fragment: CandidatesFragment): Array<TablesInsert<'candidates'>> {
    const { faker, projectId, externalIdPrefix, refs } = this.ctx;
    const rows: Array<CandidateRow> = [];

    // fixed[] pass-through — external_id prefixed, project_id defaulted
    for (const fx of fragment.fixed ?? []) {
      rows.push({ ...fx, external_id: `${externalIdPrefix}${fx.external_id}`, project_id: fx.project_id ?? projectId });
    }

    // Resolve answer emitter once per run (D-27 seam)
    const emit = this.ctx.answerEmitter ?? defaultRandomValidEmit;

    // Load questions from ctx.refs for the emitter. ctx.refs.questions carries only
    // { external_id } stubs at pipeline time — but CandidatesGenerator runs AFTER
    // QuestionsGenerator in D-06 topo order. The pipeline (Plan 07) populates
    // ctx.refs.questions with the FULL question rows after QuestionsGenerator runs
    // (see Plan 07 task for the enrichment contract). Cast guarded by that contract.
    const questionRows = refs.questions as unknown as Array<TablesInsert<'questions'>>;

    for (let i = 0; i < (fragment.count ?? 0); i++) {
      // Pick organization round-robin (deterministic — ordered by generation seed)
      const party = refs.organizations[i % Math.max(refs.organizations.length, 1)];

      const row: CandidateRow = {
        external_id: `${externalIdPrefix}cand_${String(i).padStart(4, '0')}`,
        project_id: projectId,
        first_name: faker.person.firstName(),
        last_name: faker.person.lastName()
      };

      // Organization ref (only if at least one org exists)
      if (party) {
        row.organization = { external_id: party.external_id };
      }

      // Answer emission (D-19 stub OR D-27 injected emitter)
      if (questionRows.length > 0) {
        // Build a candidate shape for the emitter. Only fields the emitter may read
        // are passed — the emitter signature is `(candidate, questions, ctx) => ...`.
        // Phase 56 default emitter doesn't use the candidate arg (_candidate);
        // Phase 57's latent emitter will read the candidate's latent position.
        const candidateForEmit: TablesInsert<'candidates'> = {
          external_id: row.external_id,
          project_id: projectId,
          first_name: row.first_name,
          last_name: row.last_name
        };
        row.answersByExternalId = emit(candidateForEmit, questionRows, this.ctx);
      }

      rows.push(row);
    }

    return rows as Array<TablesInsert<'candidates'>>;
  }
}
```

Notes:
- The `organization` ref is emitted only when `ctx.refs.organizations` has entries. If a template sets `candidates: { count: 5 }` but `organizations: { count: 0 }`, candidates are emitted WITHOUT organization — bulk_import will then fail loudly at candidate insert (CHECK `organization_id` nullable, but the candidate has no party — legal per schema but semantically broken; tests in Plan 08 should cover this edge).
- The `answersByExternalId` sentinel is emitted only when `ctx.refs.questions` has entries. If questions are absent, candidates have no answers — importAnswers will no-op (no answers collected).
- `candidateForEmit` is a narrow shape constructed for the emitter, decoupling what the emitter sees from what's actually on the row. Phase 57's latent emitter will enrich this with position/party refs.
- **IMPORTANT contract for Plan 07:** after QuestionsGenerator runs, the pipeline must populate `ctx.refs.questions` with the FULL question rows (not just external_id stubs) so CandidatesGenerator can pass them to the emitter. This is a refinement of the D-07.3 "prior-entity ref map" — for questions specifically, refs carry full rows. Plan 07's pipeline documents this.

Per RESEARCH §4.15 Pitfall 5, bulk_import's ON CONFLICT cannot match seed.sql's NULL-external_id row → UNIQUE violation on `project_id`. Route via `updateAppSettings` in the writer. The generator emits rows with `{ project_id, settings, customization }` shape; Plan 07's writer reads these, strips them from the bulk_import payload, and calls `this.client.updateAppSettings(row.settings)` for each.

```ts
import type { TablesInsert } from '@openvaa/supabase-types';
import type { Ctx, Fragment } from '../types';

export type AppSettingsFragment = Fragment<TablesInsert<'app_settings'>>;

/**
 * AppSettingsGenerator — emits app_settings blobs routed through `updateAppSettings`
 * (NOT bulk_import) per RESEARCH §4.15 Pitfall 5.
 *
 * Why: seed.sql bootstraps an app_settings row with external_id = NULL. bulk_import's
 * ON CONFLICT (project_id, external_id) WHERE external_id IS NOT NULL cannot match
 * this row — a second insert with external_id = 'seed_appsettings' falls through to
 * the UNIQUE(project_id) constraint and fails with "duplicate key value".
 *
 * Routing (Plan 07 writer):
 *   1. Pipeline emits rows from this generator
 *   2. Writer strips them from the bulk_import payload
 *   3. Writer iterates rows and calls `this.client.updateAppSettings(row.settings)` for each
 *   4. `updateAppSettings` uses the `merge_jsonb_column` RPC for deep-merging — idempotent
 *
 * Phase 56 emits at most ONE app_settings row per run (the seed.sql project has a single
 * app_settings record; merging into it is the only sensible semantics). Count > 1 is
 * silently clamped to 1 with a ctx.logger warning.
 */
export class AppSettingsGenerator {
  constructor(private ctx: Ctx) {}

  defaults(_ctx: Ctx): AppSettingsFragment {
    return { count: 0 };  // Default off — the seed.sql app_settings row is already usable
  }

  generate(fragment: AppSettingsFragment): Array<TablesInsert<'app_settings'>> {
    const { projectId, externalIdPrefix } = this.ctx;
    const rows: Array<TablesInsert<'app_settings'>> = [];

    // fixed[] pass-through — typically at most one hand-authored row
    for (const fx of fragment.fixed ?? []) {
      rows.push({ ...fx, external_id: `${externalIdPrefix}${fx.external_id}`, project_id: fx.project_id ?? projectId });
    }

    // Generated rows: emit exactly one if count >= 1 (per-project uniqueness).
    const n = Math.min(fragment.count ?? 0, 1);
    if ((fragment.count ?? 0) > 1) {
      this.ctx.logger(
        '[dev-seed] AppSettingsGenerator: app_settings is UNIQUE on project_id. ' +
        `Requested ${fragment.count} rows clamped to 1. Use one fixed[] entry per distinct project if needed.`
      );
    }
    if (n === 1 && (fragment.fixed?.length ?? 0) === 0) {
      rows.push({
        external_id: `${externalIdPrefix}appsettings`,
        project_id: projectId,
        settings: {} as unknown as TablesInsert<'app_settings'>['settings'],
        customization: {} as unknown as TablesInsert<'app_settings'>['customization']
      });
    }

    return rows;
  }
}
```
