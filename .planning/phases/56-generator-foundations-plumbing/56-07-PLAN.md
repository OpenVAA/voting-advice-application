import type { Ctx, Overrides, Template } from './types';
import { buildCtx } from './ctx';

// Generator class imports
import { AccountsGenerator } from './generators/AccountsGenerator';
import { ProjectsGenerator } from './generators/ProjectsGenerator';
import { ElectionsGenerator } from './generators/ElectionsGenerator';
import { ConstituencyGroupsGenerator } from './generators/ConstituencyGroupsGenerator';
import { ConstituenciesGenerator } from './generators/ConstituenciesGenerator';
import { OrganizationsGenerator } from './generators/OrganizationsGenerator';
import { AlliancesGenerator } from './generators/AlliancesGenerator';
import { FactionsGenerator } from './generators/FactionsGenerator';
import { QuestionCategoriesGenerator } from './generators/QuestionCategoriesGenerator';
import { QuestionsGenerator } from './generators/QuestionsGenerator';
import { CandidatesGenerator } from './generators/CandidatesGenerator';
import { NominationsGenerator } from './generators/NominationsGenerator';
import { AppSettingsGenerator } from './generators/AppSettingsGenerator';
import { FeedbackGenerator } from './generators/FeedbackGenerator';

/**
 * Topological order for generator execution.
 *
 * Source: D-06 (`bulk_import`'s `processing_order` on migration line 2751) + a
 * Phase 56 refinement: `question_categories` / `questions` run BEFORE `candidates`
 * so `ctx.refs.questions` is populated when CandidatesGenerator's answer emitter
 * (D-27 seam) iterates questions. This diverges from the DB's own processing_order
 * — which runs candidates before question_categories — because the DB doesn't care
 * about the ctx.refs.questions contract; we do.
 *
 * Accounts and projects lead the order (pass-through per D-11) so their bootstrap
 * refs are in ctx.refs before any generator that reads them (none currently — but
 * future generators may read project metadata).
 *
 * feedback runs last: direct upsert in the writer, no downstream ref consumers.
 */
export const TOPO_ORDER = [
  'accounts',
  'projects',
  'elections',
  'constituency_groups',
  'constituencies',
  'organizations',
  'alliances',
  'factions',
  'question_categories',
  'questions',     // ← before candidates per Phase 56 refinement
  'candidates',    // ← reads ctx.refs.questions for the answer emitter
  'nominations',
  'app_settings',
  'feedback'
] as const;

export type TableName = typeof TOPO_ORDER[number];

/**
 * Map from table name to generator class. Pipeline instantiates each class once
 * with ctx captured at construction (D-26), then calls `generate(fragment)`.
 * Each generator exposes `defaults(ctx): Fragment` (D-08) for smart defaults.
 */
const GENERATOR_CLASSES = {
  accounts: AccountsGenerator,
  projects: ProjectsGenerator,
  elections: ElectionsGenerator,
  constituency_groups: ConstituencyGroupsGenerator,
  constituencies: ConstituenciesGenerator,
  organizations: OrganizationsGenerator,
  alliances: AlliancesGenerator,
  factions: FactionsGenerator,
  question_categories: QuestionCategoriesGenerator,
  questions: QuestionsGenerator,
  candidates: CandidatesGenerator,
  nominations: NominationsGenerator,
  app_settings: AppSettingsGenerator,
  feedback: FeedbackGenerator
} as const;

/**
 * Run the full seeding pipeline.
 *
 * @param template Validated Template (run validateTemplate() first to surface TMPL-09 errors cleanly)
 * @param overrides Optional table → (fragment, ctx) => Rows[] map per D-25 — fully replaces built-in output
 * @param ctx Optional pre-built ctx (useful for tests that want to inject a logger or a deterministic answerEmitter)
 * @returns An object keyed by table name; each value is the array of rows that generator produced
 *
 * D-25 override signature: `(fragment, ctx) => Rows[]` — pipeline bridges with the
 * class-based built-ins per D-26:
 *   ```
 *   const gen = new Gen(ctx);
 *   const rows = overrides[table]?.(fragment, ctx) ?? gen.generate(fragment);
 *   ```
 *
 * After all generators run, a POST-TOPO ENRICHMENT PASS attaches sentinels:
 *   - `_constituencyGroups` on every election row
 *   - `_constituencies` on every constituency_group row
 *   - `_elections` on every question_category row
 *
 * These sentinels are stripped by SupabaseAdminClient.bulkImport before the RPC
 * and re-read by linkJoinTables afterwards (per RESEARCH §2 two-pass pattern).
 */
export function runPipeline(
  template: Template,
  overrides: Overrides = {},
  ctx: Ctx = buildCtx(template)
): Record<string, Array<Record<string, unknown>>> {
  const output: Record<string, Array<Record<string, unknown>>> = {};

  for (const table of TOPO_ORDER) {
    const Gen = GENERATOR_CLASSES[table];
    const gen = new Gen(ctx);
    const fragment = { ...gen.defaults(ctx), ...((template as Record<string, unknown>)[table] ?? {}) };

    // D-25 override signature + D-26 class bridge:
    // overrides[table]?.(fragment, ctx) fully replaces built-in per GEN-03 + D-05
    const rows = overrides[table]?.(fragment, ctx)
      ?? (gen.generate(fragment as never) as Array<Record<string, unknown>>);

    output[table] = rows;

    // Populate ctx.refs[table] with the FULL rows emitted. The public Ctx type in
    // ctx.ts narrows each ref to `{ external_id: string }` — the pipeline stores
    // full rows; generators that need more (CandidatesGenerator.answer emitter
    // reading question.type/choices) cast up explicitly.
    (ctx.refs as unknown as Record<string, Array<unknown>>)[table] = rows;
  }

  // Post-topo sentinel enrichment pass.
  // Plans 04 + 05 deferred sentinel emission to this pass so the generators
  // themselves stay simple and each sentinel can be populated from the final
  // ref graph (not the partial mid-topo state).
  attachSentinels(output);

  return output;
}

/**
 * Attach `_`-prefixed sentinel fields to the rows that need them.
 * bulkImport strips these before sending to the RPC; linkJoinTables reads them
 * from the SAME input dataset in a second pass.
 *
 * Sentinels:
 *   - election._constituencyGroups = { externalId: [all constituency_group extIds] }
 *   - constituency_group._constituencies = { externalId: [all constituency extIds] }
 *   - question_category._elections = { externalId: [all election extIds] }
 *
 * Phase 56 uses a "full-fanout" strategy: every election gets ALL constituency_groups,
 * every constituency_group gets ALL constituencies, every question_category gets
 * ALL elections. Phase 58 templates will let users target specific groupings.
 */
function attachSentinels(output: Record<string, Array<Record<string, unknown>>>): void {
  const allGroupExtIds = (output.constituency_groups ?? []).map((g) => g.external_id as string).filter(Boolean);
  const allConstituencyExtIds = (output.constituencies ?? []).map((c) => c.external_id as string).filter(Boolean);
  const allElectionExtIds = (output.elections ?? []).map((e) => e.external_id as string).filter(Boolean);

  if (allGroupExtIds.length > 0) {
    for (const el of output.elections ?? []) {
      (el as Record<string, unknown>)._constituencyGroups = { externalId: allGroupExtIds };
    }
  }
  if (allConstituencyExtIds.length > 0) {
    for (const cg of output.constituency_groups ?? []) {
      (cg as Record<string, unknown>)._constituencies = { externalId: allConstituencyExtIds };
    }
  }
  if (allElectionExtIds.length > 0) {
    for (const qc of output.question_categories ?? []) {
      (qc as Record<string, unknown>)._elections = { externalId: allElectionExtIds };
    }
  }
}
```

/**
 * Writer — orchestrates the three-pass write sequence per D-09 + D-10 + D-11.
 *
 * Env-var enforcement (D-15, NF-02): constructor THROWS if SUPABASE_URL or
 * SUPABASE_SERVICE_ROLE_KEY is missing. This is at construction, NOT at module
 * import — so pure generators (Wave 3) remain env-free and `yarn test:unit` can
 * exercise them without env fixture.
 *
 * NF-05 rollback (D-12): `bulk_import` runs as a single PostgREST transaction
 * (SECURITY INVOKER; migration line 2738). A mid-collection FK / constraint
 * violation aborts the RPC and nothing commits. importAnswers and linkJoinTables
 * run in SEPARATE transactions; failures there leave bulk_import rows committed —
 * acceptable per D-12. Documented here for callers.
 *
 * D-11 routing:
 *   - `accounts` / `projects` — stripped from payload (bootstrap; dev-seed does not write)
 *   - `feedback` — Phase 56 emits a logger warning and SKIPS writing (scope boundary;
 *     Phase 58 may add direct upsert if demand surfaces — feedback has no external_id
 *     so is not teardown-friendly)
 *   - `app_settings` — routed through `updateAppSettings` (RESEARCH §4.15 Pitfall 5)
 *   - 11 other tables — bulk_import → importAnswers → linkJoinTables
 */
export class Writer {
  private client: SupabaseAdminClient;
  private logger: (msg: string) => void;

  constructor(
    opts: { projectId?: string; logger?: (msg: string) => void } = {}
  ) {
    // D-15, NF-02: fail loudly before any client is constructed
    if (!process.env.SUPABASE_URL) {
      throw new Error(
        'SUPABASE_URL env var is required but not set. ' +
        'Did you forget to run `supabase start`? ' +
        'Expected format: http://127.0.0.1:54321'
      );
    }
    if (!process.env.SUPABASE_SERVICE_ROLE_KEY) {
      throw new Error(
        'SUPABASE_SERVICE_ROLE_KEY env var is required but not set. ' +
        'Run `supabase status` to obtain the local service_role key.'
      );
    }
    this.client = new SupabaseAdminClient(
      process.env.SUPABASE_URL,
      process.env.SUPABASE_SERVICE_ROLE_KEY,
      opts.projectId
    );
    this.logger = opts.logger ?? (() => {});
  }

  /**
   * Execute the full write path for a pipeline's output.
   *
   * Accepts the raw output of `runPipeline` (Record<string, Array<Record<string, unknown>>>).
   *
   * Sequence:
   *   1. Strip D-11 routing-exempt tables from the bulk payload
   *   2. bulk_import (single transaction per D-12 — 10 tables: elections, constituency_groups,
   *      constituencies, organizations, alliances, factions, candidates, question_categories,
   *      questions, nominations)
   *   3. importAnswers (stitches candidate.answersByExternalId → candidate.answers JSONB;
   *      resolves question external_ids → UUIDs)
   *   4. linkJoinTables (processes _constituencyGroups / _constituencies / _elections
   *      sentinels; writes to election_constituency_groups, constituency_group_constituencies
   *      join tables; updates question_categories.election_ids jsonb)
   *   5. updateAppSettings for each row in data.app_settings
   *   6. feedback is SKIPPED in Phase 56 with a logger warning (Claude's Discretion)
   */
  async write(data: Record<string, Array<Record<string, unknown>>>): Promise<void> {
    const bulkData: Record<string, Array<Record<string, unknown>>> = { ...data };

    // D-11: strip pass-through tables (no writes)
    delete bulkData.accounts;
    delete bulkData.projects;

    // Feedback out of bulk_import path (no external_id; Phase 56 scope-out per Claude's Discretion)
    const feedbackRows = bulkData.feedback;
    delete bulkData.feedback;

    // app_settings out of bulk_import path (RESEARCH §4.15 Pitfall 5 — UNIQUE conflict)
    const appSettingsRows = bulkData.app_settings;
    delete bulkData.app_settings;

    // Pass 1: bulk_import (10 tables, single transaction)
    await this.client.bulkImport(bulkData);

    // Pass 2: candidate answers (separate transaction)
    await this.client.importAnswers(bulkData);

    // Pass 3: join tables + category→election refs (separate transaction)
    await this.client.linkJoinTables(bulkData);

    // Pass 4: app_settings via merge_jsonb_column (Pitfall 5)
    if (appSettingsRows && appSettingsRows.length > 0) {
      for (const row of appSettingsRows) {
        const settings = row.settings;
        if (settings && typeof settings === 'object' && !Array.isArray(settings)) {
          await this.client.updateAppSettings(settings as Record<string, unknown>);
        }
      }
    }

    // Pass 5: feedback (SKIPPED in Phase 56 per scope)
    if (feedbackRows && feedbackRows.length > 0) {
      this.logger(
        `[dev-seed] Writer: feedback writes skipped in Phase 56 (${feedbackRows.length} rows ignored). ` +
        `Feedback has no external_id column and is not teardown-friendly; Phase 58 may add direct upsert support.`
      );
    }
  }
}
