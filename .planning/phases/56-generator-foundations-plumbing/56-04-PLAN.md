Canonical generator pattern (PATTERNS §generators — target shape):
```ts
import type { TablesInsert } from '@openvaa/supabase-types';
import type { Ctx, Fragment } from '../types';

export type ElectionsFragment = Fragment<TablesInsert<'elections'>>;

export class ElectionsGenerator {
  constructor(private ctx: Ctx) {}  // D-26: ctx at construction

  defaults(_ctx: Ctx): ElectionsFragment {  // D-08
    return { count: 1 };
  }

  generate(fragment: ElectionsFragment): Array<TablesInsert<'elections'>> {
    const { faker, projectId, externalIdPrefix, refs } = this.ctx;
    const rows: Array<TablesInsert<'elections'>> = [];

    // fixed[] pass-through
    for (const fixed of fragment.fixed ?? []) {
      rows.push({ ...fixed, external_id: `${externalIdPrefix}${fixed.external_id}`, project_id: projectId });
    }

    // generated rows
    for (let i = 0; i < (fragment.count ?? 0); i++) {
      rows.push({
        external_id: `${externalIdPrefix}election_${String(i).padStart(2, '0')}`,
        project_id: projectId,
        name: { en: faker.lorem.words(3) },
        election_date: faker.date.future().toISOString().slice(0, 10)
        // ... etc — each generator fills its own required/optional fields
      });
    }

    return rows;
  }
}
```

```ts
import type { TablesInsert } from '@openvaa/supabase-types';
import type { Ctx, Fragment } from '../types';

export type ConstituenciesFragment = Fragment<TablesInsert<'constituencies'>>;

// Self-FK extension: `parent: { external_id }` is a ref object, not a column.
// bulk_import's resolve_external_ref converts it to parent_id. The base row type
// doesn't model this, so we extend it.
type ConstituencyRow = TablesInsert<'constituencies'> & { parent?: { external_id: string } };

export class ConstituenciesGenerator {
  constructor(private ctx: Ctx) {}

  defaults(_ctx: Ctx): ConstituenciesFragment {
    return { count: 2 };
  }

  generate(fragment: ConstituenciesFragment): Array<TablesInsert<'constituencies'>> {
    const { faker, projectId, externalIdPrefix } = this.ctx;
    const rows: Array<ConstituencyRow> = [];

    for (const fx of fragment.fixed ?? []) {
      rows.push({ ...fx, external_id: `${externalIdPrefix}${fx.external_id}`, project_id: fx.project_id ?? projectId });
    }

    for (let i = 0; i < (fragment.count ?? 0); i++) {
      const row: ConstituencyRow = {
        external_id: `${externalIdPrefix}con_${String(i).padStart(2, '0')}`,
        project_id: projectId,
        name: { en: faker.location.state() } as unknown as TablesInsert<'constituencies'>['name'],
        sort_order: i,
        is_generated: true
      };
      // Optional self-FK: 30% of generated rows get a parent from a prior row (avoids cycles by looking backward only)
      if (rows.length > 0 && faker.number.int({ min: 1, max: 10 }) <= 3) {
        const parentIdx = faker.number.int({ min: 0, max: rows.length - 1 });
        const parentExtId = rows[parentIdx].external_id;
        if (parentExtId) row.parent = { external_id: parentExtId };
      }
      rows.push(row);
    }
    // Cast back to the declared return type — the `parent` sentinel is intentionally
    // part of the row shape consumed by bulk_import (per migration line 2640).
    return rows as Array<TablesInsert<'constituencies'>>;
  }
}
```

```ts
import type { TablesInsert } from '@openvaa/supabase-types';
import type { Ctx, Fragment } from '../types';
```

```ts
/**
 * AccountsGenerator — PASS-THROUGH per D-11.
 *
 * `accounts` is bootstrapped by `apps/supabase/supabase/seed.sql`:
 *   INSERT INTO accounts (id, name) VALUES ('00000000-0000-0000-0000-000000000001', 'Default Account');
 *
 * Dev-seed does NOT write to `accounts`. The bootstrap row is pre-populated in
 * `ctx.refs.accounts` by `buildCtx` in `packages/dev-seed/src/ctx.ts`, and
 * downstream generators reference it via `ctx.refs.accounts[0].id`.
 *
 * This class exists to:
 *   1. Satisfy GEN-01 "one generator per non-system public table" — all 16 tables.
 *   2. Make the zero-write behavior explicit (rather than silently missing).
 *   3. Let the pipeline's class map uniformly instantiate by table name.
 *
 * If a user supplies `accounts: { count: N }` or `accounts: { fixed: [...] }` in
 * a template, this generator emits a `ctx.logger` warning but still returns `[]`.
 * Writing accounts is Phase 60+ work (if ever — Supabase Auth owns account creation).
 */
export class AccountsGenerator {
  constructor(private ctx: Ctx) {}

  defaults(_ctx: Ctx): AccountsFragment {
    return { count: 0 };
  }

  generate(fragment: AccountsFragment): Array<TablesInsert<'accounts'>> {

      this.ctx.logger(

        'Requested rows ignored — bootstrap account is pre-populated via seed.sql. ' +
        `Use ctx.refs.accounts[0].id to reference it (bootstrap UUID: ${this.ctx.refs.accounts[0]?.id ?? 'none'}).`
      );
    }
    return [];
  }
}
```
