**3a. Create `voter.fixture.ts`:**

A parameterizable fixture that answers N Likert questions with a specified value. The fixture navigates through the voter app's question flow, producing a page with localStorage populated with voter answers.

```typescript
import { test as base } from '@playwright/test';
import { buildRoute } from '../utils/buildRoute';
import { testIds } from '../utils/testIds';
import type { Page } from '@playwright/test';

type VoterFixtureOptions = {
  /** Number of questions to answer. Default: all (8) */
  voterAnswerCount: number;
  /** Likert answer value index (0-based: 0=Fully disagree, 4=Fully agree). Default: 4 */
  voterAnswerIndex: number;
};

type VoterFixtures = VoterFixtureOptions & {
  /** A page with voter answers already submitted in localStorage */
  answeredVoterPage: Page;
};

export const voterTest = base.extend<VoterFixtures>({
  voterAnswerCount: [8, { option: true }],
  voterAnswerIndex: [4, { option: true }],

  answeredVoterPage: async ({ page, voterAnswerCount, voterAnswerIndex }, use) => {
    // Navigate to home page
    await page.goto(buildRoute({ route: 'Home', locale: 'en' }));

    // Click start on home page
    await page.getByTestId(testIds.voter.home.startButton).click();

    // If election selection page appears (multiple elections), select the voter election
    // Check if we landed on elections page by looking for the elections list
    const electionsListLocator = page.getByTestId(testIds.voter.elections.list);
    if (await electionsListLocator.isVisible({ timeout: 2000 }).catch(() => false)) {
      // Click the first election card (voter election)
      await page.getByTestId(testIds.voter.elections.card).first().click();
    }

    // Constituency should be auto-implied (single constituency in voter election)
    // If constituency selection appears, select the first one
    const constituenciesListLocator = page.getByTestId(testIds.voter.constituencies.list);
    if (await constituenciesListLocator.isVisible({ timeout: 2000 }).catch(() => false)) {
      await page.getByTestId(testIds.voter.constituencies.selector).first().click();
    }

    // Should be on intro page -- click start
    await page.getByTestId(testIds.voter.intro.startButton).click();

    // Answer questions
    for (let i = 0; i < voterAnswerCount; i++) {
      // Click the answer option (Likert choice at the specified index)
      await page.getByTestId(testIds.voter.questions.answerOption).nth(voterAnswerIndex).click();

      // Wait for navigation to next question (auto-advance after answer)
      // The app auto-advances after a delay. Wait for either the next question
      // or the results page (after last question)
      if (i < voterAnswerCount - 1) {
        // Wait for URL to change (next question loaded)
        await page.waitForURL(/\/questions\//);
      }
    }

    // After answering all questions, click the "Results" button (next button on last question after answering)
    await page.getByTestId(testIds.voter.questions.nextButton).click();

    // Should now be on results page
    await page.getByTestId(testIds.voter.results.list).waitFor({ state: 'visible', timeout: 10000 });

    await use(page);
  }
});
```

Note: The exact flow may need adjustment based on auto-advance behavior. The key is that after answering the last question, the next button shows "Results" and clicking it navigates to results. If auto-advance skips ahead automatically, remove the explicit next click and just wait for results.
