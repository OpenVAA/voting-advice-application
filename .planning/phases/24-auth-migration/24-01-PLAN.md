  const { error } = await this.supabase.auth.signInWithPassword({
    email: username,

  });
  if (error) throw new Error(error.message);
  return { type: 'success' as const };

protected async _logout() {
  const { error } = await this.supabase.auth.signOut({ scope: 'local' });
  if (error) throw new Error(error.message);
  return { type: 'success' as const };
}

protected async _requestForgotPasswordEmail({ email }: { email: string }) {
  const { error } = await this.supabase.auth.resetPasswordForEmail(email, {

  });
  if (error) throw new Error(error.message);
  return { type: 'success' as const };
}

protected async _resetPassword({ password }: { password: string; code: string }) {
  // Called after recovery session is established via auth callback
  // The `code` param is a Strapi-era artifact; Supabase uses the recovery session
  const { error } = await this.supabase.auth.updateUser({ password });
  if (error) throw new Error(error.message);
  return { type: 'success' as const };

protected async _setPassword({ password }: { password: string; currentPassword: string; authToken: string }) {
  // currentPassword and authToken are WithAuth compatibility shims -- ignored by Supabase

  if (error) throw new Error(error.message);
  return { type: 'success' as const };
}

Create the new auth callback route at `frontend/src/routes/[[lang=locale]]/candidate/auth/callback/+server.ts`.

This route handles Supabase auth redirects for PKCE flow (password recovery, email confirmation, invite token exchange). It receives `token_hash` and `type` as query parameters, calls `verifyOtp` on the server Supabase client (which sets session cookies via @supabase/ssr), then redirects based on the auth event type.

```typescript
import type { EmailOtpType } from '@supabase/supabase-js';
import { redirect } from '@sveltejs/kit';
import type { RequestHandler } from './$types';

export const GET: RequestHandler = async ({ url, locals }) => {
  const token_hash = url.searchParams.get('token_hash');
  const type = url.searchParams.get('type') as EmailOtpType | null;
  const next = url.searchParams.get('next');
  const lang = locals.currentLocale ?? 'en';

  if (token_hash && type) {
    const { error } = await locals.supabase.auth.verifyOtp({ token_hash, type });
    if (!error) {
      switch (type) {
        case 'recovery':
          redirect(303, `/${lang}/candidate/password-reset`);
          break;
        case 'invite':
          redirect(303, `/${lang}/candidate/register`);
          break;
        case 'email':
        case 'signup':
          redirect(303, next ? `/${lang}/${next.replace(/^\//, '')}` : `/${lang}/candidate`);
          break;
        default:
          redirect(303, `/${lang}/candidate`);
      }
    }
  }

  // Error or missing params -- redirect to login with error
  redirect(303, `/${lang}/candidate/login?errorMessage=authError`);
};
```

Key details:
- Uses `locals.supabase` (the server client created in hooks.server.ts) so session cookies are set automatically by @supabase/ssr
- Handles all Supabase auth event types: `recovery` (password reset), `invite` (candidate invite from Phase 28), `email`/`signup` (email confirmation), and a default catch-all
- Uses `locals.currentLocale` for locale-aware redirects
- Falls through to login error page if token_hash/type are missing or verifyOtp fails
- The `next` parameter allows custom redirects for email/signup confirmation flows
