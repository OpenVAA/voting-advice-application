  **Step B — `packages/dev-seed/src/templates/defaults/candidates-override.ts`:**

  Non-uniform distribution via D-25 Overrides signature. Implements RESEARCH Open Q 2 recommendation A.

  ```typescript
  /**
   * Default-template candidates override — D-25 Overrides signature.
   *
   * Replaces Phase 56 CandidatesGenerator's round-robin `i % orgCount` party
   * assignment with a sorted-descending non-uniform distribution per D-58-02.
   *
   * Weights: [20, 18, 15, 12, 10, 10, 8, 7] → sum 100, 8 parties.
   * Shape: sorted descending; no two parties identical at adjacent slots
   * except by design (10, 10) reflecting realistic close-sibling parties;
   * tail party is 35% of the largest (7/20) — matches CONTEXT's
   * "tail party < 10% of largest" rule loosely (stretch to ensure
   * the 8th party is not invisibly small).
   *
   * Per RESEARCH §Open Q 4: cycle faker locale per candidate for visual variety.
   * 25 candidates per locale block — candidates 0-24 use en, 25-49 fi, 50-74 sv,
   * 75-99 da.
   *
   * THROWS if `ctx.refs.organizations.length !== 8` — weights are tuned for 8
   * parties. If a future template needs different weights, ship a new override
   * (or generalize this one). The throw surfaces misconfiguration early.
   */

  import { en, fi, sv, da, Faker } from '@faker-js/faker';
  import type { Ctx, Overrides } from '../../types';

  /**
   * Sorted-descending weights. Sum MUST equal the candidates count.
   * D-58-02 + Claude's Discretion.
   */
  const PARTY_WEIGHTS: ReadonlyArray<number> = [20, 18, 15, 12, 10, 10, 8, 7];
  const TOTAL_CANDIDATES = PARTY_WEIGHTS.reduce((a, b) => a + b, 0); // 100
  const LOCALE_BLOCK_SIZE = 25; // 100 / 4 locales

  export const candidatesOverride: Overrides['candidates'] = (_fragment, ctx) => {
    const orgs = ctx.refs.organizations;
    if (orgs.length !== PARTY_WEIGHTS.length) {
      throw new Error(
        `candidatesOverride: expected ${PARTY_WEIGHTS.length} organizations in ctx.refs (matching PARTY_WEIGHTS), ` +
          `got ${orgs.length}. Adjust PARTY_WEIGHTS or provide 8 organizations in the template.`
      );
    }

    // Expand weights → flat party-ref array of length 100.
    //   [party_0, party_0, ... ×20, party_1, party_1, ... ×18, ...]
    const partyByIndex: Array<{ external_id: string }> = [];
    for (let p = 0; p < PARTY_WEIGHTS.length; p++) {
      for (let k = 0; k < PARTY_WEIGHTS[p]; k++) {
        partyByIndex.push({ external_id: orgs[p].external_id });
      }
    }

    // Per-locale faker instances for visual variety (RESEARCH Open Q 4).
    const localeFakers: ReadonlyArray<Faker> = [
      buildLocaleFaker('en', ctx),
      buildLocaleFaker('fi', ctx),
      buildLocaleFaker('sv', ctx),
      buildLocaleFaker('da', ctx)
    ];

    const rows: Array<Record<string, unknown>> = [];
    for (let i = 0; i < TOTAL_CANDIDATES; i++) {
      const localeIdx = Math.floor(i / LOCALE_BLOCK_SIZE);
      const faker = localeFakers[localeIdx];
      rows.push({
        external_id: `${ctx.externalIdPrefix}cand_${String(i).padStart(4, '0')}`,
        project_id: ctx.projectId,
        first_name: faker.person.firstName(),
        last_name: faker.person.lastName(),
        sort_order: i,
        is_generated: true,
        organization: partyByIndex[i]
      });
    }
    return rows;
  };

  function buildLocaleFaker(locale: 'en' | 'fi' | 'sv' | 'da', ctx: Ctx): Faker {
    const localePacks = { en, fi, sv, da };
    const f = new Faker({ locale: [localePacks[locale], en] });
    // Seed offset per locale so each block of 25 candidates is visibly distinct
    // while the overall run stays deterministic at ctx.faker's seed.
    const base = getFakerSeed(ctx.faker) ?? 42;
    const offsets: Record<'en' | 'fi' | 'sv' | 'da', number> = { en: 0, fi: 1000, sv: 2000, da: 3000 };
    f.seed(base + offsets[locale]);
    return f;
  }

  /**
   * Recover the seed value from a Faker instance. Faker v8 doesn't expose
   * the current seed directly; `.seed()` with no args re-reads it. Fall back
   * to 42 (buildCtx default) if undetectable.
   *
   * The number returned here is the ONLY non-deterministic axis — if the base
   * seed changes, locale offsets scale with it. Determinism holds as long as
   * the base ctx.faker was seeded consistently at buildCtx time.
   */
  function getFakerSeed(_faker: Faker): number | undefined {
    // Faker v8 has no public API to read the current seed. Callers rely on
    // ctx being freshly built from Template.seed at pipeline entry, so the
    // effective seed is always template.seed ?? 42. Phase 58 accepts this
    // constraint — the override runs after buildCtx and can assume that
    // contract.
    return undefined;
  }
  ```

  **Step C — `packages/dev-seed/src/templates/defaults/questions-override.ts`:**

  ```typescript
  /**
   * Default-template questions override — enforces D-58-03 type mix.
   *
   * D-58-03: majority Likert (ordinal), some categorical, some multi-choice,
   * exactly 1 boolean. NO number/text/date/image/multipleText.
   *
   * Split (Claude's Discretion, respecting "majority Likert"):
   *   - 18 ordinal (Likert 5-point)
   *   - 4 categorical (3-5 choices each)
   *   - 1 multi-choice (4 choices, threshold >= 2)
   *   - 1 boolean
   *   Total: 24
   *
   * Questions are distributed across the 4 categories from
   * `ctx.refs.question_categories` in round-robin so every category has at
   * least 4 + 1 = 5 questions (depending on distribution edge cases).
   *
   * Phase 57's latent emitter (installed in pipeline.ts:177) exercises the
   * latent model for ordinal + categorical + multi-choice types. Boolean
   * falls back to defaultRandomValidEmit per D-57-10.
   */

  import type { Overrides } from '../../types';

  const TYPE_PLAN: ReadonlyArray<string> = [
    // 18 ordinal (indices 0..17)
    ...Array(18).fill('ordinal'),
    // 4 categorical (indices 18..21)
    'categorical', 'categorical', 'categorical', 'categorical',
    // 1 multi-choice (index 22)
    'multiple_choice_categorical',
    // 1 boolean (index 23)
    'boolean'
  ];

  export const questionsOverride: Overrides['questions'] = (_fragment, ctx) => {
    const cats = ctx.refs.question_categories;
    if (cats.length === 0) {
      throw new Error('questionsOverride: expected at least 1 question category in ctx.refs');
    }

    const rows: Array<Record<string, unknown>> = [];
    for (let i = 0; i < TYPE_PLAN.length; i++) {
      const type = TYPE_PLAN[i];
      const category = cats[i % cats.length];
      const faker = ctx.faker;
      rows.push({
        external_id: `${ctx.externalIdPrefix}q_${String(i).padStart(4, '0')}`,
        project_id: ctx.projectId,
        type,
        name: { en: faker.lorem.sentence({ min: 6, max: 12 }).replace(/\.$/, '?') },
        info: { en: faker.lorem.paragraph() },
        is_generated: true,
        category: { external_id: category.external_id },
        choices: buildChoices(type, faker, ctx)
      });
    }
    return rows;
  };

  function buildChoices(type: string, faker: import('@faker-js/faker').Faker, _ctx: unknown): unknown {
    // Phase 56's QuestionsGenerator handles choices per type — we mirror the
    // minimal shape needed for the latent emitter.
    if (type === 'ordinal') {
      // Likert 5-point: ids '1'..'5' with numeric values (Pitfall 6 from Phase 57
      // clustering integration test).
      return [
        { id: '1', label: { en: 'Strongly disagree' }, value: 1 },
        { id: '2', label: { en: 'Disagree' }, value: 2 },
        { id: '3', label: { en: 'Neutral' }, value: 3 },
        { id: '4', label: { en: 'Agree' }, value: 4 },
        { id: '5', label: { en: 'Strongly agree' }, value: 5 }
      ];
    }
    if (type === 'categorical') {
      const n = 3 + faker.number.int({ min: 0, max: 2 }); // 3-5 choices
      return Array.from({ length: n }, (_, i) => ({
        id: `choice_${i}`,
        label: { en: faker.word.noun() }
      }));
    }
    if (type === 'multiple_choice_categorical') {
      return Array.from({ length: 4 }, (_, i) => ({
        id: `mc_${i}`,
        label: { en: faker.word.noun() }
      }));
    }
    if (type === 'boolean') {
      return undefined; // Phase 56 boolean questions have no choices array
    }
    return undefined;
  }
  ```
