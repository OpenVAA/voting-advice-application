2. **Create `frontend/src/lib/supabase/server.ts`** -- factory for creating per-request Supabase server client:
```typescript
import { PUBLIC_SUPABASE_URL, PUBLIC_SUPABASE_ANON_KEY } from '$env/static/public';
import { createServerClient } from '@supabase/ssr';
import type { Database } from '@openvaa/supabase-types';
import type { RequestEvent } from '@sveltejs/kit';

/**
 * Create a Supabase server client with cookie-based auth.
 * Call this once per request in hooks.server.ts and attach to event.locals.
 */
export function createSupabaseServerClient(event: RequestEvent) {
  return createServerClient<Database>(
    PUBLIC_SUPABASE_URL,
    PUBLIC_SUPABASE_ANON_KEY,
    {
      cookies: {
        getAll: () => event.cookies.getAll(),
        setAll: (cookiesToSet) => {
          cookiesToSet.forEach(({ name, value, options }) => {
            event.cookies.set(name, value, { ...options, path: '/' });
          });
        }
      }
    }
  );
}
```

import { PUBLIC_SUPABASE_URL, PUBLIC_SUPABASE_ANON_KEY } from '$env/static/public';
import { createBrowserClient } from '@supabase/ssr';
import type { Database } from '@openvaa/supabase-types';

let browserClient: ReturnType<typeof createBrowserClient<Database>> | null = null;

/**
 * Get or create the browser-side Supabase client.
 * Singleton -- safe to call multiple times.
 */
export function createSupabaseBrowserClient() {
  if (browserClient) return browserClient;
  browserClient = createBrowserClient<Database>(
    PUBLIC_SUPABASE_URL,
    PUBLIC_SUPABASE_ANON_KEY
  );
  return browserClient;
}

The updated handle function should:
```typescript
export const handle: Handle = (async ({ event, resolve }) => {
  // Create Supabase server client FIRST (before any other logic)
  const supabase = createSupabaseServerClient(event);

  // Define safeGetSession
  const safeGetSession = async () => {
    const { data: { session } } = await supabase.auth.getSession();
    if (!session) return { session: null, user: null };
    const { data: { user }, error } = await supabase.auth.getUser();
    if (error) return { session: null, user: null };
    return { session, user };
  };

  // ... existing locale handling (sections 1-3) unchanged ...

  // Section 4: Handle candidate requests
  // Keep existing AUTH_TOKEN_KEY check for backward compat
  // Also check Supabase session as alternative
  if (pathname.startsWith(`/${servedLocale}/candidate`)) {
    const token = event.cookies.get(AUTH_TOKEN_KEY);
    // Check both old Strapi token and new Supabase session
    const hasAuth = token || event.cookies.getAll().some(c => c.name.startsWith('sb-'));

    if (hasAuth && pathname.endsWith('candidate/login')) {
      redirect(303, `/${servedLocale}/candidate`);
    }
    if (!hasAuth && route.id.includes('(protected)')) {
      redirect(303, `/${servedLocale}/candidate/login?redirectTo=${cleanPath.substring(1)}`);
    }
  }

  // Section 5: Serve content with Supabase client on locals
  return resolve(
    {
      ...event,
      locals: {
        supabase,
        safeGetSession,
        currentLocale: servedLocale,
        preferredLocale
      }
    },
    {
      transformPageChunk: ({ html }) => html.replace('%lang%', `${servedLocale}`),
      filterSerializedResponseHeaders(name) {
        return name === 'content-range' || name === 'x-supabase-api-version';
      }
    }
  );
});
```

Keep `handleError` and `debug` functions unchanged.
