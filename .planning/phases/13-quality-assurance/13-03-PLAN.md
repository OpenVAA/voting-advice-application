<!-- Storage path convention from 014-storage.sql -->
Path format: {project_id}/{entity_type}/{entity_id}/filename.ext
- (storage.foldername(name))[1] = project_id
- (storage.foldername(name))[2] = entity_type (e.g., 'candidates', 'organizations')
- (storage.foldername(name))[3] = entity_id

Buckets: 'public-assets' (public=true), 'private-assets' (public=false)

Storage policies:
- Anon SELECT public-assets: only if is_storage_entity_published(type, id) = true
- Authenticated SELECT: published OR can_access_project OR own entity
- Candidate INSERT/UPDATE/DELETE: own entity folder in 'candidates' type only
- Admin INSERT/UPDATE/DELETE: any file in their project

<!-- bulk_import/bulk_delete from 016-bulk-operations.sql -->
Both are SECURITY INVOKER -- RLS policies apply to the caller.
bulk_import(data jsonb) -> jsonb: expects {"collection_name": [{...items...}]}
bulk_delete(data jsonb) -> jsonb: expects {"project_id": "uuid", "collections": {"name": {"prefix"|"ids"|"external_ids": ...}}}

<!-- Column-level protections from 013-auth-rls.sql -->
candidates: REVOKE UPDATE FROM authenticated, then GRANT UPDATE (name, short_name, info, color, image, sort_order, subtype, custom_data, first_name, last_name, answers, created_at, updated_at)
Protected columns: published, project_id, auth_user_id, organization_id, id, is_generated

organizations: REVOKE UPDATE FROM authenticated, then GRANT UPDATE (name, short_name, info, color, image, sort_order, subtype, custom_data, answers, created_at, updated_at)
Protected columns: published, project_id, auth_user_id, id, is_generated

<!-- Triggers from 000-functions.sql and 015-external-id.sql -->
validate_answer_value(answer_val jsonb, q_type question_type, valid_choices jsonb) -> void (raises exception on invalid)
validate_nomination() -> trigger (enforces hierarchy + election/constituency consistency)
enforce_external_id_immutability() -> trigger (NULL->value ok, value->different blocked)
validate_answers_jsonb() -> trigger on candidates/organizations (validates JSONB answers column)
