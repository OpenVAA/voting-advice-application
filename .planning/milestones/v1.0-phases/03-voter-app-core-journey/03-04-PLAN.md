import voterDataset from '../../data/voter-dataset.json' assert { type: 'json' };

**Pre-compute expected rankings in module scope:**

```typescript
const QUESTION_COUNT = 8;
const VOTER_ANSWER_VALUE = 5; // "Fully agree"

// Create questions matching the voter dataset's 8 Likert-5 questions
const questions = voterDataset.questions.map(q =>
  OrdinalQuestion.fromLikert({ id: q.externalId, scale: 5 })
);

// Create voter answers (all "Fully agree")
const voterAnswers: Record<string, { value: number }> = {};
for (const q of questions) {
  voterAnswers[q.id] = { value: VOTER_ANSWER_VALUE };
}

// Filter to visible candidates (those with termsOfUseAccepted)
const visibleCandidates = voterDataset.candidates.filter(c => c.termsOfUseAccepted);

// Build HasAnswers-compatible objects for each candidate
const candidateEntities = visibleCandidates.map(c => {
  const answers: Record<string, { value: number }> = {};
  if (c.answersByExternalId) {
    for (const [qId, ans] of Object.entries(c.answersByExternalId)) {
      if (questions.some(q => q.id === qId) && ans.value) {
        answers[qId] = { value: Number(ans.value) };
      }
    }
  }
  return { name: `${c.firstName} ${c.lastName}`, externalId: c.externalId, answers };
});

// Run matching algorithm (same config as frontend voterContext.ts)
const algorithm = new MatchingAlgorithm({
  distanceMetric: DISTANCE_METRIC.Manhattan,
  missingValueOptions: { method: MISSING_VALUE_METHOD.RelativeMaximum }
});
```

Note: The `match()` API may require objects implementing `HasAnswers` interface from `@openvaa/core`. If plain objects don't work, create a simple wrapper class. Check the actual API and adapt.
