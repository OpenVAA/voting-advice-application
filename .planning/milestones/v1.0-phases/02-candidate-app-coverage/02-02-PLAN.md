test.describe('candidate registration via email', () => {
  // This describe block must be serial because steps depend on each other
  test.describe.configure({ mode: 'serial' });

  const client = new StrapiAdminClient();
  const candidateEmail = 'test.unregistered@openvaa.org';
  let registrationLink: string;

  test.beforeAll(async () => {
    await client.login();
  });

  test.afterAll(async () => {
    await client.dispose();
  });

  test('should send registration email and extract link', async () => {
    // Step 1: Find the unregistered candidate's documentId via client.findData
    const findResult = await client.findData('candidates', {
      email: { $eq: candidateEmail }
    });
    const candidateDocumentId = findResult.data?.[0]?.documentId;
    expect(candidateDocumentId).toBeTruthy();

    // Step 2: Send registration email via client.sendEmail
    await client.sendEmail({
      candidateId: candidateDocumentId!,
      subject: 'Registration',
      content: 'Click here to register: {LINK}',
      requireRegistrationKey: true
    });

    // Step 3: Poll for email arrival using expect.poll (NOT waitForTimeout)
    const html = await expect.poll(
      async () => await getLatestEmailHtml(candidateEmail),
      { message: 'Waiting for registration email', timeout: 15000, intervals: [1000, 2000, 3000] }
    ).toBeTruthy();

    // Step 4: Extract registration link
    // Note: expect.poll returns the value but typed as void. Refetch after assertion:
    const emailHtml = await getLatestEmailHtml(candidateEmail);
    const link = extractLinkFromHtml(emailHtml!);
    expect(link).toBeTruthy();
    registrationLink = link!;
  });

  test('should complete registration via email link', async ({ page, registerPage }) => {
    // Step 1: Navigate to the registration link extracted from email
    await page.goto(registrationLink);

    // Step 2: Set password using registerPage
    await registerPage.setPassword('RegisteredPass1!', 'RegisteredPass1!');

    // Step 3: Expect auto-login (URL should navigate away from register page to home)
    // Wait for URL to change to candidate home or similar
    await expect(page).toHaveURL(/candidate/);
    // Verify candidate home status message is visible
  });
});
