### Pattern: split conditional test into 2 tests (RESEARCH §"Pattern 4" canonical 1)

```typescript
// BEFORE
test('foo', async ({ page }) => {
  if (someCondition) await bar();
  else await baz();
});

// AFTER — describe-block with `if` at module/describe scope (NOT inside the test fn)
test.describe('foo cases', () => {
  if (someCondition) {
    test('foo (cond branch)', async ({ page }) => { await bar(); });
  } else {
    test('foo (else branch)', async ({ page }) => { await baz(); });
  }
});
```

Use when the two branches represent distinct test contracts (e.g., multilocale vs single-locale; auth-enabled vs auth-disabled).

### Pattern: hoist setup to beforeEach (RESEARCH §"Pattern 4" canonical 2)

```typescript
// BEFORE
test('bar', async ({ page }) => {
  switch (mode) { case 'single': await generateOne(); break; case 'multi': await generateMany(); break; }
  await expect(page.locator('.my-image')).toBeVisible();
});

// AFTER
test.beforeEach(async () => {
  switch (mode) { case 'single': await generateOne(); break; case 'multi': await generateMany(); break; }
});
test('bar', async ({ page }) => {
  await expect(page.locator('.my-image')).toBeVisible();
});
```

Use when the conditional is BRANCHY SETUP, not branchy assertions.

### Pattern: hoist to module-level (RESEARCH §"Pattern 4" canonical 3)

```typescript
// BEFORE
test('baz', async ({ page }) => {
  const hotkey = process.platform === 'linux' ? ['Control','Alt','f'] : ['Alt','f'];
  await page.keyboard.press(hotkey.join('+'));
});

// AFTER
const hotkey = process.platform === 'linux' ? ['Control','Alt','f'] : ['Alt','f'];
test('baz', async ({ page }) => {
  await page.keyboard.press(hotkey.join('+'));
});
```

Use for true platform/env switches (rare in this codebase per Plan 1 inventory).

### Pattern: getError wrapper for try-catch error capture (RESEARCH §"Pattern 5")

```typescript
// BEFORE
test('includes status code', async () => {
  try {
    await makeRequest(url);
  } catch (error) {
    expect(error).toHaveProperty('statusCode', 404);  // ← never runs if no throw → false pass
  }
});

// AFTER
class NoErrorThrownError extends Error {}
const getError = async <TError>(call: () => unknown): Promise<TError> => {
  try { await call(); throw new NoErrorThrownError(); }
  catch (e) { return e as TError; }
};
test('includes status code', async () => {
  const error = await getError(async () => makeRequest(url));
  expect(error).not.toBeInstanceOf(NoErrorThrownError);
  expect(error).toHaveProperty('statusCode', 404);
});
```

Use only for try-catch sites (rare; most no-conditional-expect sites in this codebase are `if (await x.isVisible()) expect(...)` — those collapse to `await expect(x).toBeVisible(); expect(...);` per Pitfall 8 + Pattern 5 commentary).
