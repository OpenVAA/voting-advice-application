  expect(normalizeCoordinate({ value: -10, min: -10, max: 10 })).toEqual(COORDINATE.Min);
});

Deno.test("normalizeCoordinate - max maps to COORDINATE.Max", () => {
  expect(normalizeCoordinate({ value: 10, min: -10, max: 10 })).toEqual(COORDINATE.Max);
});

Deno.test("normalizeCoordinate - midpoint between neutral and max", () => {
  expect(normalizeCoordinate({ value: 5, min: -10, max: 10 })).toEqual(

});

Create `packages/core/tests_deno/getEntity.test.ts` with this exact content:
```typescript
import { describe, it } from "@std/testing/bdd";
import { expect } from "@std/expect";
import { getEntity } from "../src/entity/getEntity.ts";
import type { Entity } from "../src/entity/entity.type.ts";

const ENTITY: Entity = { answers: {} };

describe("getEntity", () => {
  it("should return the naked entity when given a plain object", () => {
    const result = getEntity(ENTITY);
    expect(result).toBe(ENTITY);
  });

  it("should return the entity from a WrappedEntity object", () => {
    const result = getEntity({ entity: ENTITY });
    expect(result).toBe(ENTITY);
  });

  it("should return the target from a WrappedEntity and MatchedEntity object", () => {
    const result = getEntity({ target: { entity: ENTITY } });
    expect(result).toBe(ENTITY);
  });

  it("should return the target from a MatchedEntity object", () => {
    const result = getEntity({ target: ENTITY });
    expect(result).toBe(ENTITY);
  });

  it("should throw an error when given a null value", () => {
    expect(() => getEntity(null as unknown as Entity)).toThrow();
  });

  it("should throw an error when given a primitive value", () => {
    expect(() => getEntity("not an entity" as unknown as Entity)).toThrow();
  });
});
```

IMPORTANT: All imports from source use `.ts` extensions (e.g., `../src/matching/missingValue.ts` NOT `../src/matching/missingValue`). Deno requires explicit file extensions.

The getEntity test uses `describe`/`it` from `@std/testing/bdd` to demonstrate both Deno.test and BDD-style patterns work. The other two use plain `Deno.test` for simplicity.
