**Step 1: Bulk regex migration of $t() -> t() across all .svelte files.**

The primary migration is mechanical: drop the `$` store subscription prefix. In Svelte templates and script blocks:

```bash
# In all .svelte files under apps/frontend/src/:
# Replace {$t( with {t(  -- template expressions
# Replace ($t( with (t(  -- function call arguments
# Replace =$t( with =t(  -- attribute assignments
# Replace :$t( with :t(  -- ternary expressions
# Replace ,$t( with ,t(  -- array/object literals
# Replace !$t( with !t(  -- negation
# Replace  $t( with  t(  -- space-separated (catch-all)
```

The pattern is: any `$t(` that is a translation call (not part of a variable name like `$theme`) becomes `t(`. The `$t` symbol is specifically the sveltekit-i18n store subscription.

Use a comprehensive regex replacement:
```

```

import { getContext, hasContext, setContext } from 'svelte';
import { t, translate, getLocale, locales } from '$lib/i18n';
import type { I18nContext } from './i18nContext.type';

const CONTEXT_KEY = Symbol();

export function getI18nContext() {
  if (!hasContext(CONTEXT_KEY)) error(500, 'GetI18nContext() called before initI18nContext()');
  return getContext<I18nContext>(CONTEXT_KEY);

export function initI18nContext(): I18nContext {
  if (hasContext(CONTEXT_KEY)) error(500, 'InitI18nContext() called for a second time');

}

const translationLocales = fs.readdirSync(messagesDir).filter((name) => {

/**
 * Recursive function to extract leaf keys, handling inlang variant arrays.
 * Variant arrays (array values) are treated as leaf nodes (same as string values).
 */
function flattenKeys(obj: unknown, prefix: string): string[] {
  const res: string[] = [];
  if (typeof obj !== 'object' || obj === null || Array.isArray(obj)) {
    // Leaf node (string, number, or variant array)
    res.push(prefix);
  } else {
    for (const [key, value] of Object.entries(obj as Record<string, unknown>)) {
      res.push(...flattenKeys(value, prefix ? `${prefix}.${key}` : key));
    }
  }
  return res.sort();
}

function getMessageKeys(locale: string, filename: string): string[] {
  const filePath = path.join(messagesDir, locale, filename);
  const content = JSON.parse(fs.readFileSync(filePath, 'utf8'));
  return flattenKeys(content, filename.replace('.json', ''));
}
