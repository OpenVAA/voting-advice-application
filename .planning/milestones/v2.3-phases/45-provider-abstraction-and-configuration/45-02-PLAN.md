```typescript
export interface AuthConfig {
  identityMatchProp: string;
  extractClaims: string[];
  firstNameProp: string;
  lastNameProp: string;

  readonly type: ProviderType;
  readonly authConfig: AuthConfig;
  getAuthorizeUrl(params: AuthorizeParams): Promise<AuthorizeResult>;
  exchangeCodeForToken(params: TokenExchangeParams): Promise<TokenExchangeResult>;
  getIdTokenClaims(idToken: string): Promise<IdTokenClaimsResult>;
}

export interface AuthorizeParams {
  redirectUri: string;
  codeVerifier?: string;
  codeChallenge?: string;
}

export interface AuthorizeResult {
  authorizeUrl: string;
  clientSideRedirect: boolean;
}

export interface TokenExchangeParams {
  authorizationCode: string;
  redirectUri: string;
  codeVerifier?: string;
}

export interface TokenExchangeResult {
  idToken: string;
}

export type IdTokenClaimsResult =
  | { success: true; data: { firstName: string; lastName: string; identifier: string; extractedClaims: Record<string, string> } }
```

- Read `IDENTITY_PROVIDER_TOKEN_ENDPOINT` and `IDENTITY_PROVIDER_CLIENT_SECRET` from server constants
- Read `PUBLIC_IDENTITY_PROVIDER_CLIENT_ID` from publicConstants
- POST to token endpoint with URLSearchParams body:

  - `redirect_uri`: redirectUri

  - `client_secret`: IDENTITY_PROVIDER_CLIENT_SECRET
- If response not ok, throw `new Error('Token exchange failed')`
- Parse response JSON, extract `id_token`
- Return `{ idToken: id_token }`

- Replicate the exact logic from existing `getIdTokenClaims.ts` but use `authConfig` for claim extraction:
- Parse `IDENTITY_PROVIDER_DECRYPTION_JWKS` from server constants into `jose.JWK[]`
- `jose.decodeProtectedHeader(idToken)` to get kid

- `jose.compactDecrypt(idToken, await jose.importJWK(matchingJwk))` to decrypt JWE
- `jose.jwtVerify(decryptedText, jose.createRemoteJWKSet(new URL(constants.IDENTITY_PROVIDER_JWKS_URI)), { audience: publicConstants.PUBLIC_IDENTITY_PROVIDER_CLIENT_ID, issuer: constants.IDENTITY_PROVIDER_ISSUER })` to verify JWT

  - `identifier`: `payload[this.authConfig.identityMatchProp]` cast to string
  - `extractedClaims`: Object built from `this.authConfig.extractClaims.map(claim => [claim, String(payload[claim] ?? '')])` into `Record<string, string>` via `Object.fromEntries()`
- Return success result with data object, or catch errors and return failure result matching existing pattern

```typescript
export function getActiveProvider(): IdentityProvider {
  const providerType = (constants.PUBLIC_IDENTITY_PROVIDER_TYPE || 'signicat') as ProviderType;
  switch (providerType) {
    case 'idura':
      return iduraProvider;
    case 'signicat':
      return signicatProvider;
    default:
      throw new Error(`Unknown identity provider type: ${providerType}. Expected 'signicat' or 'idura'.`);
  }
}
```
